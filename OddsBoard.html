#!/usr/bin/env python3 """ Run this script to update OddsBoard HTML files: python3 update_oddsboard.py OddsBoard-Dark.html python3 update_oddsboard.py OddsBoard-Light.html """ import sys, os def transform(html): # Step 1: Protect SHOW -> placeholder reps1 = [ ('pool-chip show', 'CHIP__SHOOTOUT__PH'), ('pool-show', 'POOL__SHOOTOUT__PH'), ('.show-cell', '.CELL__SHOOTOUT__PH'), ('.show-odds', '.ODDS__SHOOTOUT__PH'), ('show-cell', 'CELL__SHOOTOUT__PH'), ('show-odds', 'ODDS__SHOOTOUT__PH'), ('show-dim', 'DIM__SHOOTOUT__PH'), ('show-bg', 'BG__SHOOTOUT__PH'), ('--show', '--SHOOTOUT__PH'), ('pools.show', 'pools.SHOOTOUT__PH'), ('showOdds', 'SHOOTOUT__ODDS__PH'), ('showBets', 'SHOOTOUT__BETS__PH'), ("'SHOW $'", "'SHOOTOUT__DOLLAR__PH'"), ('"SHOW $"', '"SHOOTOUT__DOLLAR__PH"'), ('>SHOW $<', '>SHOOTOUT__DOLLAR__PH<'), ("+ 'SHOW $'", "+ 'SHOOTOUT__DOLLAR__PH'"), (">SHOW<", ">SHOOTOUT__UC__PH<"), (">Show<", ">SHOOTOUT__TC__PH<"), ("'Show'", "'SHOOTOUT__TC__PH'"), ('"Show"', '"SHOOTOUT__TC__PH"'), ('showNoBets', 'SHOOTOUT__NOBETS__PH'), ] for old, new in reps1: html = html.replace(old, new) # Step 2: PLACE -> SHOW reps2 = [ ('pool-chip place', 'pool-chip show'), ('pool-place', 'pool-show'), ('.place-cell', '.show-cell'), ('.place-odds', '.show-odds'), ('place-cell', 'show-cell'), ('place-odds', 'show-odds'), ('place-dim', 'show-dim'), ('place-bg', 'show-bg'), ('--place', '--show'), ('pools.place', 'pools.show'), ('placeOdds', 'showOdds'), ('placeBets', 'showBets'), ("'PLACE $'", "'SHOW $'"), ('"PLACE $"', '"SHOW $"'), ('>PLACE $<', '>SHOW $<'), ("+ 'PLACE $'", "+ 'SHOW $'"), (">PLACE<", ">SHOW<"), (">Place<", ">Show<"), ("'Place'", "'Show'"), ('"Place"', '"Show"'), ('placeNoBets', 'showNoBets'), ] for old, new in reps2: html = html.replace(old, new) # Step 3: Restore placeholders -> SHOOT-OUT reps3 = [ ('CHIP__SHOOTOUT__PH', 'pool-chip shootout'), ('POOL__SHOOTOUT__PH', 'pool-shootout'), ('.CELL__SHOOTOUT__PH', '.shootout-cell'), ('.ODDS__SHOOTOUT__PH', '.shootout-odds'), ('CELL__SHOOTOUT__PH', 'shootout-cell'), ('ODDS__SHOOTOUT__PH', 'shootout-odds'), ('DIM__SHOOTOUT__PH', 'shootout-dim'), ('BG__SHOOTOUT__PH', 'shootout-bg'), ('--SHOOTOUT__PH', '--shootout'), ('pools.SHOOTOUT__PH', 'pools.shootout'), ('SHOOTOUT__ODDS__PH', 'shootoutOdds'), ('SHOOTOUT__BETS__PH', 'shootoutBets'), ("'SHOOTOUT__DOLLAR__PH'", "'SHOOT-OUT $'"), ('"SHOOTOUT__DOLLAR__PH"', '"SHOOT-OUT $"'), ('>SHOOTOUT__DOLLAR__PH<', '>SHOOT-OUT $<'), ("+ 'SHOOTOUT__DOLLAR__PH'", "+ 'SHOOT-OUT $'"), (">SHOOTOUT__UC__PH<", ">SHOOT-OUT<"), (">SHOOTOUT__TC__PH<", ">Shoot-Out<"), ("'SHOOTOUT__TC__PH'", "'Shoot-Out'"), ('"SHOOTOUT__TC__PH"', '"Shoot-Out"'), ('SHOOTOUT__NOBETS__PH', 'shootoutNoBets'), ] for old, new in reps3: html = html.replace(old, new) # Also fix CSS var declarations in :root if present html = html.replace('--place: #3B82F6', '--show: #3B82F6') html = html.replace('--show: #A855F7', '--shootout: #A855F7') # And CSS rules html = html.replace( '.pool-chip.place { color: var(--place); border-color: #1e3a8a; background: #0a1628; }', '.pool-chip.show { color: var(--show); border-color: #1e3a8a; background: #0a1628; }' ) html = html.replace( '.pool-chip.show { color: var(--show); border-color: #581c87; background: #160d22; }', '.pool-chip.shootout { color: var(--shootout); border-color: #581c87; background: #160d22; }' ) html = html.replace( '.odds-cell.place-cell { color: var(--place); border-right: 1px solid; }', '.odds-cell.show-cell { color: var(--show); border-right: 1px solid; }' ) html = html.replace( '.odds-cell.show-cell { color: var(--show); }', '.odds-cell.shootout-cell { color: var(--shootout); }' ) return html if len(sys.argv) < 2: print("Usage: python3 update_oddsboard.py ") sys.exit(1) filepath = sys.argv[1] with open(filepath, 'r', encoding='utf-8', errors='replace') as f: src = f.read() out = transform(src) outpath = filepath.replace('.html', '_updated.html') with open(outpath, 'w', encoding='utf-8') as f: f.write(out) print(f"Written: {outpath}") # Verify checks_zero = ['pool-place','place-cell','place-odds','place-dim','place-bg','--place', 'pools.place','placeOdds','placeBets'] checks_present = ['pool-show','pool-shootout','show-cell','shootout-cell', 'showOdds','shootoutOdds','showBets','shootoutBets','SHOW $','SHOOT-OUT $'] print("\n--- Should be 0 (old terms) ---") for t in checks_zero: c = out.count(t) print(f" {c} {t} {'OK' if c==0 else 'PROBLEM'}") print("\n--- Should be > 0 (new terms) ---") for t in checks_present: c = out.count(t) print(f" {c} {t} {'OK' if c>0 else 'MISSING'}")
